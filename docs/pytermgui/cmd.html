 27class AppWindow(ptg.Window):
 28    """A generic application window.
 29
 30    It contains a header with the app's title, as well as some global
 31    settings.
 32    """
 33
 34    app_title: str
 35    """The display title of the application."""
 36
 37    app_id: str
 38    """The short identifier used by ArgumentParser."""
 39
 40    standalone: bool
 41    """Whether this app was launched directly from the CLI."""
 42
 43    overflow = ptg.Overflow.SCROLL
 44    vertical_align = ptg.VerticalAlignment.TOP
 45
 46    def __init__(self, args: Namespace | None = None, **attrs: Any) -> None:
 47        super().__init__(**attrs)
 48
 49        self.standalone = bool(getattr(args, self.app_id, None))
 50
 51        bottom = ptg.Container.chars["border"][-1]
 52        header_box = ptg.boxes.Box(
 53            [
 54                "",
 55                " x ",
 56                bottom * 3,
 57            ]
 58        )
 59
 60        self._add_widget(ptg.Container(f"[ptg.title]{self.app_title}", box=header_box))
 61        self._add_widget("")
 62
 63    def setup(self) -> None:
 64        """Centers window, sets its width & height."""
 65
 66        self.width = int(self.terminal.width * 2 / 3)
 67        self.height = int(self.terminal.height * 2 / 3)
 68        self.center(store=False)
 69
 70    def on_exit(self) -> None:
 71        """Called on application exit.
 72
 73        Should be used to print current application state to the user's shell.
 74        """
 75
 76        ptg.tim.print(f"{_title()} - [dim]{self.app_title}")
 77        print()

 80class GetchWindow(AppWindow):
 81    """A window for the Getch utility."""
 82
 83    app_title = "Getch"
 84    app_id = "getch"
 85
 86    def __init__(self, args: Namespace | None = None, **attrs: Any) -> None:
 87        super().__init__(args, **attrs)
 88
 89        self.bind(ptg.keys.ANY_KEY, self._update)
 90
 91        self._content = ptg.Container("Press any key...", static_width=50)
 92        self._add_widget(self._content)
 93
 94        self.setup()
 95
 96    def _update(self, _: ptg.Widget, key: str) -> None:
 97        """Updates window contents on keypress."""
 98
 99        self._content.set_widgets([])
100        name = _get_key_name(key)
101
102        if name != ascii(key):
103            name = f"keys.{name}"
104
105        style = ptg.HighlighterStyle(ptg.highlight_python)
106
107        items = [
108            "[ptg.title]Your output",
109            "",
110            {"[ptg.detail]key": ptg.Label(name, style=style)},
111            {"[ptg.detail]value:": ptg.Label(ascii(key), style=style)},
112            {"[ptg.detail]len()": ptg.Label(str(len(key)), style=style)},
113            {
114                "[ptg.detail]real_length()": ptg.Label(
115                    str(ptg.real_length(key)), style=style
116                )
117            },
118        ]
119
120        for item in items:
121            self._content += item
122
123        if self.standalone:
124            assert self.manager is not None
125            self.manager.stop()
126
127    def on_exit(self) -> None:
128        super().on_exit()
129
130        for line in self._content.get_lines():
131            print(line)

134class ColorPickerWindow(AppWindow):
135    """A window to pick colors from the xterm-256 palette."""
136
137    app_title = "ColorPicker"
138    app_id = "color"
139
140    def __init__(self, args: Namespace | None = None, **attrs: Any) -> None:
141        super().__init__(args, **attrs)
142
143        self._chosen_rgb = ptg.str_to_color("black")
144
145        self._colorpicker = ptg.ColorPicker()
146        self._add_widget(ptg.FancyReprWidget(ptg.palette, starts_at=2))
147        self._add_widget(ptg.Collapsible("xterm-256", "", self._colorpicker))
148        self._add_widget("")
149        self._add_widget(
150            ptg.Collapsible("RGB & HEX", "", self._create_rgb_picker(), static_width=81)
151        )

155    def _create_rgb_picker(self) -> ptg.Container:
156        """Creates the RGB picker 'widget'."""
157
158        root = ptg.Container(static_width=72)
159
160        matrix = ptg.DensePixelMatrix(68, 20)
161        hexdisplay = ptg.Label()
162        rgbdisplay = ptg.Label()
163
164        sliders = [ptg.Slider() for _ in range(3)]
165
166        def _get_rgb() -> tuple[int, int, int]:
167            """Computes the RGB value from the 3 sliders."""
168
169            values = [int(255 * slider.value) for slider in sliders]
170
171            return values[0], values[1], values[2]
172
173        def _update(*_) -> None:
174            """Updates the matrix & displays with the current color."""
175
176            color = self._chosen_rgb = ptg.RGBColor.from_rgb(_get_rgb())
177            for row in range(matrix.rows):
178                for col in range(matrix.columns):
179                    matrix[row, col] = color.hex
180
181            hexdisplay.value = f"[ptg.body]{color.hex}"
182            rgbdisplay.value = f"[ptg.body]rgb({', '.join(map(str, color.rgb))})"
183            matrix.build()
184
185        red, green, blue = sliders
186
187        # red.styles.filled_selected__cursor = "red"
188        # green.styles.filled_selected__cursor = "green"
189        # blue.styles.filled_selected__cursor = "blue"
190
191        for slider in sliders:
192            slider.onchange = _update
193
194        root += hexdisplay
195        root += rgbdisplay
196        root += ""
197
198        root += matrix
199        root += ""
200
201        root += red
202        root += green
203        root += blue
204
205        _update()
206        return root
207
208    def on_exit(self) -> None:
209        super().on_exit()
210
211        color = self._chosen_rgb
212        eightbit = " ".join(
213            button.get_lines()[0] for button in self._colorpicker.chosen
214        )
215
216        ptg.tim.print("[ptg.title]Your colors:")
217        ptg.tim.print(f"    [{color.hex}]{color.rgb}[/] // [{color.hex}]{color.hex}")
218        ptg.tim.print()
219        ptg.tim.print(f"    {eightbit}")

284    @staticmethod
285    def _create_showcase() -> ptg.Container:
286        """Creates the showcase container."""
287
288        def _show_style(name: str) -> str:
289            return f"[{name}]{name}"  # .replace("'", "")
290
291        def _create_table(source: Iterable[tuple[str, str]]) -> ptg.Container:
292            root = ptg.Container()
293
294            for left, right in source:
295                row = ptg.Splitter(
296                    ptg.Label(left, parent_align=0), ptg.Label(right, parent_align=2)
297                )
298
299                row.set_char("separator", f" {ptg.Container.chars['border'][0]}")
300
301                root += row
302
303            return root
304
305        prefix = "ptg.timwindow"
306        tags = [_show_style(style) for style in ptg.markup.style_maps.STYLES]
307        colors = [
308            f"[[{prefix}.255]0-255[/]]",
309            f"[[{prefix}.hex]#RRGGBB[/]]",
310            f"[[{prefix}.rgb]RRR;GGG;BBB[/]]",
311            "",
312            f"[[inverse {prefix}.255]@0-255[/]]",
313            f"[[inverse {prefix}.hex]@#RRGGBB[/]]",
314            f"[[inverse {prefix}.rgb]@RRR;GGG;BBB[/]]",
315        ]
316
317        tag_container = _create_table(zip_longest(tags, colors, fillvalue=""))
318        user_container = _create_table(
319            (_show_style(tag), f"[{tag}]{value}")
320            for tag, value in ptg.tim.aliases.items()
321            if not tag.startswith("/") and tag not in ptg.palette.data
322        )
323
324        return ptg.Container(tag_container, user_container, box="EMPTY")

361    @staticmethod
362    def obj_from_path(path: str) -> object | None:
363        """Retrieves an object from any valid import path.
364
365        An import path could be something like:
366            pytermgui.window_manager.compositor.Compositor
367
368        ...or if the library in question imports its parts within `__init__.py`-s:
369            pytermgui.Compositor
370        """
371
372        parts = path.split(".")
373
374        if parts[0] in dir(builtins):
375            obj = getattr(builtins, parts[0])
376
377        elif parts[0] in dir(ptg):
378            obj = getattr(ptg, parts[0])
379
380        else:
381            try:
382                obj = importlib.import_module(".".join(parts[:-1]))
383            except (ValueError, ModuleNotFoundError) as error:
384                return (
385                    f"Could not import object at path {path!r}: {error}."
386                    + " Maybe try using the --eval flag?"
387                )
388
389        try:
390            obj = getattr(obj, parts[-1])
391        except AttributeError:
392            return obj
393
394        return obj

396    def _update(self, *_) -> None:

399        obj = self.obj_from_path(self._input.value)

553def screenshot(man: ptg.WindowManager) -> None:
554    """Opens a modal dialogue & saves a screenshot."""
555
556    tempname = ".screenshot_temp.svg"
557
558    modal: ptg.Window
559
560    def _finish(*_: Any) -> None:
561        """Closes the modal and renames the window."""
562
563        man.remove(modal)
564        filename = field.value or "screenshot"
565
566        if not filename.endswith(".svg"):
567            filename += ".svg"
568
569        os.rename(tempname, filename)
570
571        man.toast("[ptg.title]Screenshot saved!", "", f"[ptg.detail]{filename}")

578    modal = man.alert(
579        "[ptg.title]Screenshot taken!", "", ptg.Container(field), "", ["Save!", _finish]

609def _create_app_picker(manager: ptg.WindowManager) -> ptg.Window:
610    """Creates a dropdown that allows picking between applications."""
611
612    existing_windows: list[ptg.Window] = []
613
614    def _wrap(func: Callable[[ptg.Widget], Any]) -> Callable[[ptg.Widget], Any]:
615        def _inner(caller: ptg.Widget) -> None:
616            dropdown.collapse()
617
618            window: ptg.Window = func(caller)
619            if type(window) in map(type, manager):
620                return
621
622            existing_windows.append(window)
623            manager.add(window, assign="body")
624
625            body = manager.layout.body
626
627            body.content = window
628            manager.layout.apply()
629
630        return _inner
631
632    buttons = [
633        ptg.Button(label, _wrap(lambda *_, app=app: app()))
634        for (label, _), app in APPLICATION_MAP.items()
635    ]
636
637    dropdown = ptg.Collapsible("Applications", *buttons, keyboard=True).styles(
638        fill="ptg.header"
639    )
640
641    return ptg.Window(
642        dropdown,
643        box="EMPTY",
644        id="ptg.header",
645        is_persistent=True,
646        overflow=ptg.Overflow.RESIZE,
647    ).styles(fill="ptg.header")

650def _create_footer(man: ptg.WindowManager) -> ptg.Window:
651    """Creates a footer based on the manager's bindings."""
652
653    content = ptg.Splitter().styles(fill="ptg.footer")
654    for key, (callback, doc) in man.bindings.items():
655        if doc == f"Binding of {key} to {callback}":
656            continue
657
658        content.lazy_add(
659            ptg.Button(
660                f"{_get_key_name(str(key))} - {doc}",
661                onclick=lambda *_, _callback=callback: _callback(man),
662            )
663        )
664
665    return ptg.Window(content, box="EMPTY", id="ptg.footer", is_persistent=True)

716def run_environment(args: Namespace) -> None:
717    """Runs the WindowManager environment.
718
719    Args:
720        args: An argparse namespace containing relevant arguments.
721    """
722
723    def _find_focused(manager: ptg.WindowManager) -> ptg.Window | None:
724        if manager.focused is None:
725            return None
726
727        # Find foremost non-persistent window
728        for window in manager:
729            if window.is_persistent:
730                continue
731
732            return window
733
734        return None
735
736    def _toggle_attachment(manager: ptg.WindowManager) -> None:
737        focused = _find_focused(manager)
738
739        if focused is None:
