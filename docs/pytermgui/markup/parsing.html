 93def consume_tag(tag: str) -> Token:  # pylint: disable=too-many-return-statements
 94    """Consumes a tag text, returns the associated Token."""
 95
 96    if tag in STYLES:
 97        return StyleToken(tag)
 98
 99    if tag.startswith("/"):
100        return ClearToken(tag)
101
102    if tag.startswith("!"):
103        matchobj = RE_MACRO.match(tag)
104
105        if matchobj is not None:
106            name, args = matchobj.groups()
107
108            if args is None:
109                return MacroToken(name, tuple())
110
111            return MacroToken(name, tuple(args.split(":")))
112
113    if tag.startswith("~"):
114        return HLinkToken(tag[1:])
115
116    if tag.startswith("(") and tag.endswith(")"):
117        values = tag[1:-1].split(";")
118        if len(values) != 2:
119            raise MarkupSyntaxError(
120                tag,
121                f"should have exactly 2 values separated by `;`, not {len(values)}",
122                "",
123            )
124
125        return CursorToken(tag[1:-1], *map(int, values))
126
127    if tag in PSEUDO_TOKENS:
128        return PseudoToken(tag)
129
130    token: Token
131    try:
132        token = ColorToken(tag, Color.parse(tag, localize=False))
133
134    except ColorSyntaxError:
135        token = AliasToken(tag)
136
137    finally:
138        return token  # pylint: disable=lost-exception

141def tokenize_markup(text: str) -> Iterator[Token]:
142    """Converts some markup text into a stream of tokens.
143
144    Args:
145        text: Any valid markup.
146
147    Yields:
148        The generated tokens, in the order they occur within the markup.
149    """
150
151    cursor = 0
152    length = len(text)
153    has_inverse = False
154    for matchobj in RE_MARKUP.finditer(text):
155        full, escapes, content = matchobj.groups()
156        start, end = matchobj.span()
157
158        if cursor < start:
159            yield PlainToken(text[cursor:start])
160
161        if not escapes == "":
162            _, remaining = divmod(len(escapes), 2)
163
164            yield PlainToken(full[max(1 - remaining, 1) :])
165            cursor = end
166
167            continue
168
169        for tag in content.split():
170            if tag == "inverse":
171                has_inverse = True
172
173            if tag == "/inverse":
174                has_inverse = False
175
176            consumed = consume_tag(tag)
177            if has_inverse:
178                if consumed.markup == "/fg":
179                    consumed = ClearToken("/fg")
180
181                elif consumed.markup == "/bg":
182                    consumed = ClearToken("/bg")
183
184            yield consumed
185
186        cursor = end
187
188    if cursor < length:
189        yield PlainToken(text[cursor:length])

192def tokenize_ansi(  # pylint: disable=too-many-locals, too-many-branches, too-many-statements
193    text: str,
194) -> Iterator[Token]:
195    """Converts some ANSI-coded text into a stream of tokens.
196
197    Args:
198        text: Any valid ANSI-coded text.
199
200    Yields:
201        The generated tokens, in the order they occur within the text.
202    """
203
204    cursor = 0
205
206    for matchobj in RE_ANSI.finditer(text):
207        start, end = matchobj.span()
208
209        csi = matchobj.groups()[0:2]
210        link_osc = matchobj.groups()[2:4]
211
212        if cursor < start:
213            yield PlainToken(text[cursor:start])
214
215        if link_osc != (None, None):
216            cursor = end
217            uri, label = link_osc
218
219            yield HLinkToken(uri)
220            yield PlainToken(label)
221            yield ClearToken("/~")
222
223            continue
224
225        full, content = csi
226
227        cursor = end
228
229        code = ""
230
231        # Position
232        posmatch = RE_POSITION.match(full)
233
234        if posmatch is not None:
235            ypos, xpos = posmatch.groups()
236            if not ypos and not xpos:
237                raise ValueError(
238                    f"Cannot parse cursor when no position is supplied. Match: {posmatch!r}"
239                )
240
241            yield CursorToken(content, int(ypos) or None, int(xpos) or None)
242            continue
243
244        parts = content.split(";")
245
246        state = None
247        color_code = ""
248        for part in parts:
249            if state is None:
250                if part in REVERSE_STYLES:
251                    yield StyleToken(REVERSE_STYLES[part])
252                    continue
253
254                if part in REVERSE_CLEARERS:
255                    yield ClearToken(REVERSE_CLEARERS[part])
256                    continue
257
258                if part in ("38", "48"):
259                    state = "COLOR"
260                    color_code += part + ";"
261                    continue
262
263                # standard colors
264                try:
265                    yield ColorToken(part, Color.parse(part, localize=False))
266                    continue
267
268                except ColorSyntaxError as exc:
269                    raise ValueError(f"Could not parse color tag {part!r}.") from exc
270
271            if state != "COLOR":
272                continue
273
274            color_code += part + ";"
275
276            # Ignore incomplete RGB colors
277            if (
278                color_code.startswith(("38;2;", "48;2;"))
279                and len(color_code.split(";")) != 6
280            ):
281                continue
282
283            try:
284                code = color_code
285
286                if code.startswith(("38;2;", "48;2;", "38;5;", "48;5;")):
287                    stripped = code[5:-1]
288
289                    if code.startswith("4"):
290                        stripped = "@" + stripped
291
292                    code = stripped
293
294                yield ColorToken(code, Color.parse(code, localize=False))
295
296            except ColorSyntaxError:
297                continue
298
299            state = None
300            color_code = ""
301
302    remaining = text[cursor:]
303    if len(remaining) > 0:
304        yield PlainToken(remaining)

307def eval_alias(text: str, context: ContextDict) -> str:
308    """Evaluates a space-delimited string of alias tags into their underlying value.
309
310    Args:
311        text: A space-separated string containing the aliases.
312
313    Returns:
314        The space-separated string that the input aliases represent.
315    """
316
317    aliases = context["aliases"]
318
319    evaluated = ""
320    for tag in text.split():
321        if tag not in aliases:
322            evaluated += tag + " "
323            continue
324
325        evaluated += eval_alias(aliases[tag], context) + " "
326
327    return evaluated.rstrip(" ")

350def parse_macro(
351    token: MacroToken, context: ContextDict, get_full: Callable[[], str]
352) -> tuple[MacroType, tuple[str, ...]]:
353    """Parses a macro token.
354
355    Returns:
356        A tuple containing the callable bound to the name, as well as the arguments
357        passed to it.
358    """
359
360    func = context["macros"].get(token.value)
361
362    if func is None:
363        dump = json.dumps(context["macros"], indent=2, default=str)
364
365        raise MarkupSyntaxError(
366            token.value, f"not defined in macro context: {dump}", get_full()
367        )
368
369    return func, token.arguments

372def parse_alias(
373    token: AliasToken, context: ContextDict, get_full: Callable[[], str]
374) -> str:
375    """Parses an alias token."""
376
377    if token.value not in context["aliases"]:
378        dump = json.dumps(context["aliases"], indent=2, default=str)
379
380        raise MarkupSyntaxError(
381            token.value, f"not defined in alias context: {dump}", get_full()
382        )
383
384    meaning = context["aliases"][token.value]
385
386    return eval_alias(meaning, context).rstrip(" ")

389def parse_clear(token: ClearToken, _: ContextDict, get_full: Callable[[], str]) -> str:
390    """Parses a clearer token."""
391
392    if token.value == "/~":
393        return "\x1b]8;;\x1b\\"
394
395    index = CLEARERS.get(token.value)
396    if index is None:
397        raise MarkupSyntaxError(
398            token.value, "not a recognized clearer or alias", get_full()
399        )
400
401    return f"\x1b[{index}m"

412def parse_state_pseudo(
413    token: PseudoToken,
414    tokens: list[Token],
415    index: int,
416    save_state: list[Token],
417    context: ContextDict,
418) -> str:
419    """Parses a state pseudo tokens"""
420
421    tag = token.value
422
423    parsed = ""
424
425    if tag.startswith(STATE_CUT):
426        save_state.clear()
427        save_state.extend(filter(lambda tkn: not tkn.is_plain(), tokens[:index]))
428
429        if not tag == STATE_COPY:
430            parsed += "\x1b[0m"
431
432    # Restore
433    else:
434        local_state = save_state.copy()
435
436        if tag == STATE_REPLACE:
437            local_state.insert(0, ClearToken("/"))
438
439        parsed = parse_tokens(
440            local_state,
441            context=context,
442            optimize=False,
443            append_reset=False,
444        )
445
446    return parsed

449def optimize_tokens(tokens: list[Token]) -> Iterator[Token]:
450    """Optimizes a stream of tokens, only yielding functionally relevant ones.
451
452    Args:
453        tokens: Any list of Token objects. Usually obtained from `tokenize_markup`
454            or `tokenize_ansi`.
455
456    Yields:
457        All those tokens within the input iterator that are functionally relevant,
458            keeping their order.
459    """
460
461    previous: list[Token] = []
462    current_tag_group: list[Token] = []
463
464    def _diff_previous() -> Iterator[Token]:
465        """Find difference from the previously active list of tokens."""
466
467        applied = previous.copy()
468
469        for tkn in current_tag_group:
470            targets = []
471
472            clearer = Token.is_clear(tkn)
473            if Token.is_clear(tkn):
474                targets = [tkn.targets(tag) for tag in applied]
475
476            if tkn in previous and not clearer:
477                continue
478
479            if clearer and not any(targets):
480                continue
481
482            applied.append(tkn)
483            yield tkn
484
485    def _remove_redundant_color(token: Token) -> None:
486        """Removes non-functional colors.
487
488        These happen in the following ways:
489        - Multiple colors of the same channel (fg/bg) are present.
490        - A color is applied, then a clearer clears it.
491        """
492
493        for applied in current_tag_group.copy():
494            if Token.is_clear(applied) and applied.targets(token):
495                current_tag_group.remove(applied)
496
497            if not Token.is_color(applied):
498                continue
499
500            old = applied.color
501
502            if old.background == new.background:
503                current_tag_group.remove(applied)
504
505    for token in tokens:
506        if Token.is_plain(token):
507            yield from _diff_previous()
508            yield token
509
510            previous = current_tag_group.copy()
511
512            continue
513
514        if Token.is_color(token):
515            new = token.color
516
517            _remove_redundant_color(token)
518
519            if not any(token.markup == applied.markup for applied in current_tag_group):
520                current_tag_group.append(token)
521
522            continue
523
524        if token.is_style():
525            if not any(token == tag for tag in current_tag_group):
526                current_tag_group.append(token)
527
528            continue
529
530        if Token.is_clear(token):
531            applied = False
532            for tag in current_tag_group.copy():
533                if token.targets(tag) or token == tag:
534                    current_tag_group.remove(tag)
535                    applied = True
536
537            if not applied:
538                continue
539
540        current_tag_group.append(token)
541
542    yield from _diff_previous()

545def tokens_to_markup(tokens: list[Token]) -> str:
546    """Converts a token stream into the markup of its tokens.
547
548    Args:
549        tokens: Any list of Token objects. Usually obtained from `tokenize_markup` or
550            `tokenize_ansi`.
551
552    Returns:
553        The markup the given tokens represent.
554    """
555
556    tags: list[Token] = []
557    markup = ""
558
559    for token in tokens:
560        if token.is_plain():
561            if len(tags) > 0:
562                markup += f"[{' '.join(tag.markup for tag in tags)}]"
563
564            markup += token.value
565
566            tags = []
567
568        else:
569            tags.append(token)
570
571    if len(tags) > 0:
572        markup += f"[{' '.join(tag.markup for tag in tags)}]"
573
574    return markup

600def _apply_macros(
601    text: str, macros: Iterator[tuple[MacroType, tuple[str, ...]]]
602) -> str:
603    """Applies macros to the given text.
604
605    Args:
606        text: The plain text the macros will apply to.
607        macros: Any iterator of MacroTokens that will be applied.
608
609    Returns:
610        The input plain text, with all macros applied to it. The macros will be applied
611        in the order they appear in.
612    """
613
614    for method, args in macros:
615        if len(args) > 0:
616            text = method(*args, text)
617            continue
618
619        text = method(text)
620
621    return text

624def _sub_aliases(tokens: list[Token], context: ContextDict) -> list[Token]:
625    """Substitutes all AliasTokens to their underlying values.
626
627    Args:
628        tokens: Any list of Tokens. When this iterator contains nothing
629            that can be interpreted as an alias, the same iterator turned into
630            a list will be returned.
631        context: The context that aliases will be searched in.
632    """
633
634    output: list[Token] = []
635
636    # It's more computationally efficient to create this lambda once and reuse it
637    # every time. There is no need to define a full function, as it just returns
638    # a function return.
639    get_full = (
640        lambda: tokens_to_markup(  # pylint: disable=unnecessary-lambda-assignment
641            tokens
642        )
643    )
644
645    for token in tokens:
646        if token.value in context["aliases"] and (
647            Token.is_clear(token) or Token.is_macro(token) or Token.is_alias(token)
648        ):
649            if Token.is_clear(token) or Token.is_macro(token):
650                token = AliasToken(token.value)
651
652            if Token.is_alias(token):
653                aliases_parsed = parse_alias(token, context, get_full)
654                output.extend(list(tokenize_markup(f"[{aliases_parsed}]")))
655
656            continue
657
658        if Token.is_macro(token) and token.value == "!link":
659            warn(
660                "Hyperlinks are no longer implemented as macros."
661                + " Prefer using the `~{uri}` syntax.",
662                DeprecationWarning,
663                stacklevel=4,
664            )
665
666            output.append(HLinkToken(":".join(token.arguments)))
667            continue
668
669        output.append(token)
670
671    return output

675def parse_tokens(  # pylint: disable=too-many-branches, too-many-locals, too-many-statements
676    tokens: list[Token],
677    *,
678    optimize: bool = False,
679    context: ContextDict | None = None,
680    append_reset: bool = True,
681    ignore_unknown_tags: bool = True,
682) -> str:
683    """Parses a stream of tokens into the ANSI-coded string they represent.
684
685    Args:
686        tokens: Any list of Tokens, usually obtained from either `tokenize_ansi` or
687            `tokenize_markup`.
688        optimize: If set, `optimize_tokens` will optimize the input iterator before
689            usage. This will incur a (minor) performance hit.
690        context: The context that aliases and macros found within the tokens will be
691            searched in.
692        append_reset: If set, `ClearToken("/")` will be appended to the token iterator,
693            clearing all styles.
694        ignore_unknown_tags: If set, the `MarkupSyntaxError` coming from unknown tags
695            will be silenced.
696
697    Returns:
698        The ANSI-coded string that the token stream represents.
699    """
700
701    if context is None:
702        context = create_context_dict()
703
704    token_list = _sub_aliases(tokens, context)
705
706    # It's more computationally efficient to create this lambda once and reuse it
707    # every time. There is no need to define a full function, as it just returns
708    # a function return.
709    get_full = (
710        lambda: tokens_to_markup(  # pylint: disable=unnecessary-lambda-assignment
711            token_list
712        )
713    )
714
715    if optimize:
716        token_list = list(optimize_tokens(token_list))
717
718    if append_reset:
719        token_list.append(ClearToken("/"))
720
721    link = None
722    output = ""
723    segment = ""
724    background = Color.parse("#000000")
725    macros: list[MacroToken] = []
726    unknown_aliases: list[Token] = []
727
728    save_state: list[Token] = []
729
730    for i, token in enumerate(token_list):
731        if token.is_plain():
732            value = _apply_macros(
733                token.value, (parse_macro(macro, context, get_full) for macro in macros)
734            )
735
736            if len(unknown_aliases) > 0:
737                output += f"[{' '.join(tkn.value for tkn in unknown_aliases)}]"
738                unknown_aliases = []
739
740            output += segment + (
741                value if link is None else LINK_TEMPLATE.format(uri=link, label=value)
742            )
743
744            segment = ""
745            continue
746
747        if token.is_hyperlink():
748            link = token.value
749            continue
750
751        if Token.is_macro(token):
752            macros.append(token)
753            continue
754
755        if Token.is_clear(token):
756            if token.value in ("/", "/~"):
757                link = None
758
759                if token.value == "/~":
760                    continue
761
762            found = False
763            for macro in macros.copy():
764                if token.targets(macro):
765                    macros.remove(macro)
766                    found = True
767                    break
768
769            if found and token.value != "/":
770                continue
771
772            if token.value.startswith("/!"):
773                raise MarkupSyntaxError(
774                    token.value, "has nothing to target", get_full()
775                )
776
777        if Token.is_color(token) and token.color.background:
778            background = token.color
779
780        if Token.is_pseudo(token):
781            if token.value in STATE_PSEUDOS:
782                segment += parse_state_pseudo(token, tokens, i, save_state, context)
783                continue
784
785            if token.value == "#auto":
786                token = ColorToken("#auto", background.contrast)
787
788        try:
789            segment += PARSERS[type(token)](token, context, get_full)  # type: ignore
790
791        except MarkupSyntaxError:
792            if not ignore_unknown_tags:
793                raise
794
795            unknown_aliases.append(token)
796
797    if len(unknown_aliases) > 0:
798        output += f"[{' '.join(tkn.value for tkn in unknown_aliases)}]"
799
800    output += segment
801
802    return output

805def parse(
806    text: str,
807    optimize: bool = False,
808    context: ContextDict | None = None,
809    append_reset: bool = True,
810    ignore_unknown_tags: bool = True,
811) -> str:
812    """Parses markup into the ANSI-coded string it represents.
813
814    Args:
815        text: Any valid markup.
816        optimize: If set, `optimize_tokens` will optimize the tokens found within the
817            input markup before usage. This will incur a (minor) performance hit.
818        context: The context that aliases and macros found within the markup will be
819            searched in.

451    """Optimizes a stream of tokens, only yielding functionally relevant ones.
452
453    Args:

465    def _diff_previous() -> Iterator[Token]:
466        """Find difference from the previously active list of tokens."""
467
468        applied = previous.copy()
469
470        for tkn in current_tag_group:
471            targets = []

496                current_tag_group.remove(applied)

499                continue
