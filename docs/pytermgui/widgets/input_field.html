208    def get_word_pos(self, direction: Literal[-1, 1]) -> int:
209        """Gets the column offset to the next word in the given direction.
210
211        Args:
212            direction: Which direction we need to look for.
213
214        Returns:
215            The column offset.
216        """
217
218        row, col = self.cursor
219        if len(self._lines) <= row:
220            return direction
221
222        # Consistent with unix shell behaviour:
223        # * Always delete first char, then remove any non-punctuation
224        # Note that the exact behaviour isn't standardized:
225        # * Python repl: until change in letter+digit & punctionation
226        # * Unix shells: only removes letter+digit
227        word_chars = string.ascii_letters + string.digits
228
229        if direction == -1:
230            line = self._lines[row][: col - 1]
231            strip_line = line.rstrip(word_chars)
232
233        else:
234            line = self._lines[row][col:]
235            strip_line = line.lstrip(word_chars)
236
237        return -direction * (len(strip_line) - len(line)) + direction

239    def handle_action(self, action: str) -> bool:
240        """Handles some action.
241
242        This will be expanded in the future to allow using all behaviours with
243        just their actions.
244        """
245
246        cursors = {
247            "move_left": (0, -1),
248            "move_right": (0, 1),
249            "move_up": (-1, 0),
250            "move_down": (1, 0),
251        }
252
253        if action.startswith("move_"):
254            if action.endswith(("word_left", "word_right")):
255                col = self.get_word_pos(-1 if action == "move_word_left" else 1)
256                self.move_cursor((0, col))
257                return True
258
259            if action.endswith(("end", "home")):
260                crow, ccol = self.cursor
261                if action == "move_end":
262                    ccol = len(self._lines[crow])
263                else:
264                    ccol = 0
265                self.move_cursor((crow, ccol), absolute=True)
266                return True
267
268            row, col = cursors[action]
269
270            if self.cursor.row + row > len(self._lines):
271                self._lines.append("")
272
273            col += self._selection_length
274            if self._selection_length > 0:
275                col -= 1
276
277            self._selection_length = 1
278            self.move_cursor((row, col))
279            return True
280
281        if action.startswith("select_"):
282            if action == "select_right":
283                self.update_selection(1)
284
285            elif action == "select_left":
286                self.update_selection(-1)
287
288            return True
289
290        if action == "word_remove":
291            row, col = self.cursor
292            self.delete_back(-self.get_word_pos(-1))
293            return True
294
295        return False

298    def handle_key(  # pylint: disable=too-many-return-statements, too-many-branches
299        self, key: str
300    ) -> bool:
301        """Adds text to the field, or moves the cursor."""
302
303        if self.execute_binding(key, ignore_any=True):
304            return True
305
306        for name, options in self.keys.items():
307            if (
308                name.rsplit("_", maxsplit=1)[-1] in ("up", "down")
309                and not self.multiline
310            ):
311                continue
312
313            if key in options:
314                return self.handle_action(name)
315
316        if key == keys.TAB:
317            if not self.multiline:
318                return False
319
320            for _ in range(self.tablength):
321                self.handle_key(" ")
322
323            return True
324
325        if key in string.printable and key not in "\x0c\x0b":
326            if key == keys.ENTER:
327                if not self.multiline:
328                    return False
329
330                if len(self._lines) <= self.cursor.row:
331                    self._lines.append("")
332
333                line = self._lines[self.cursor.row]
334                left, right = line[: self.cursor.col], line[self.cursor.col :]
335
336                self._lines[self.cursor.row] = left
337                self._lines.insert(self.cursor.row + 1, right)
338
339                self.move_cursor((1, -self.cursor.col))
340                self._styled_cache = None
341
342            else:
343                self.insert_text(key)
344
345            if keys.ANY_KEY in self._bindings:
346                method, _ = self._bindings[keys.ANY_KEY]
347                method(self, key)
348
349            return True
350
351        if key == keys.BACKSPACE:
352            if self._selection_length == 1:
353                self.delete_back(1)
354            else:
355                self.delete_back(-self._selection_length)
356
357            self._selection_length = 1
358            self._styled_cache = None
359
360            return True
361
362        if len(key) > 1 and not key.startswith("\x1b["):
363            for char in key:
364                self.handle_key(char)
365
366            return True
367
368        return False

370    def handle_mouse(self, event: MouseEvent) -> bool:
371        """Allows point-and-click selection."""
372
373        x_offset = event.position[0] - self.pos[0]
374        y_offset = event.position[1] - self.pos[1]
375
376        if y_offset == 0:
377            x_offset -= len(self.prompt)
378
379            if x_offset < 0:
380                return False
381
382        # Set cursor to mouse location
383        if event.action is MouseAction.LEFT_CLICK:
384            if not y_offset < len(self._lines):
385                return False
386
387            line = self._lines[y_offset]
388
389            if y_offset == 0:
390                line = self.prompt + line
391
392            self.move_cursor((y_offset, min(len(line), x_offset)), absolute=True)
393
394            self._drag_start = (x_offset, y_offset)
395            self._selection_length = 1
396
397            return True
398
399        # Select text using dragging the mouse
400        if event.action is MouseAction.LEFT_DRAG and self._drag_start is not None:
401            change = x_offset - self._drag_start[0]
402            self.update_selection(
403                change - self._selection_length + 1, correct_zero_length=False
404            )
405
406            return True
407
408        return super().handle_mouse(event)

410    def move_cursor(self, new: tuple[int, int], *, absolute: bool = False) -> None:
411        """Moves the cursor, then possible re-positions it to a valid location.
412
413        Args:
414            new: The new set of (y, x) positions to use.
415            absolute: If set, `new` will be interpreted as absolute coordinates,
416                instead of being added on top of the current ones.
417        """
418
419        if len(self._lines) == 0:
420            return
421
422        if absolute:
423            new_y, new_x = new
424            self.cursor.row = new_y
425            self.cursor.col = new_x
426
427        else:
428            self.cursor += new
429
430        self.cursor.row = max(0, min(self.cursor.row, len(self._lines) - 1))
431        row, col = self.cursor
432
433        line = self._lines[row]
434
435        # Going left, possibly upwards
436        if col < 0:
437            if row <= 0:
438                self.cursor.col = 0
439
440            else:
441                self.cursor.row -= 1
442                line = self._lines[self.cursor.row]
443                self.cursor.col = len(line)
444
445        # Going right, possibly downwards
446        elif col > len(line) and line != "":
447            if len(self._lines) > row + 1:
448                self.cursor.row += 1
449                self.cursor.col = 0
450
451            line = self._lines[self.cursor.row]
452
453        self.cursor.col = max(0, min(self.cursor.col, len(line)))
454
455    def get_lines(self) -> list[str]:
456        """Builds the input field's lines."""
457
458        if not self._cache_is_valid() or self._styled_cache is None:
459            self._styled_cache = self._style_and_break_lines()
460
461        lines = self._styled_cache
462
463        row, col = self.cursor
464
465        if len(self._lines) == 0:
466            line = " "
467        else:
468            line = self._lines[row]
469
470        start = col
471        cursor_char = " "
472        if len(line) > col:
473            start = col
474            end = col + self._selection_length
475            start, end = sorted([start, end])
476
477            try:
478                cursor_char = line[start:end]
479            except IndexError as error:
480                raise ValueError(f"Invalid index in {line!r}: {col}") from error
481
482        style_cursor = (
483            self.styles.value if self.selected_index is None else self.styles.cursor
484        )
485
486        # TODO: This is horribly hackish, but is the only way to "get around" the
487        #       limits of the current scrolling techniques. Should be refactored
488        #       once a better solution is available
489        if self.parent is not None and self.selected_index is not None:
490            offset = 0
491            parent = self.parent
492            while hasattr(parent, "parent"):
493                offset += getattr(parent, "_scroll_offset")
494
495                parent = parent.parent  # type: ignore
496
497            offset_row = -offset + row
498            offset_col = start + (len(self.prompt) if row == 0 else 0)
499
500            if offset_col > self.width - 1:
501                offset_col -= self.width
502                offset_row += 1
503                row += 1
504
505                if row >= len(lines):
506                    lines.append(self.styles.value(""))
507
508            position = (
509                self.pos[0] + offset_col,
510                self.pos[1] + offset_row,
511            )
512
513            self.positioned_line_buffer.append(
514                (position, style_cursor(cursor_char))  # type: ignore
515            )
516
517        lines = lines or [""]
518        self.height = len(lines)
519
520        return lines
